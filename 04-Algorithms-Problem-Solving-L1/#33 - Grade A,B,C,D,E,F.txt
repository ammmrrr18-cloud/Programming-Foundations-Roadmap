#include<iostream>
using namespace std;

int ReadNumberInRange(int From, int To)
{
	int Grade;

	do
	{
		cout << "Please enter your grade between 0 and 100 ?\n";
		cin >> Grade;

	} while (Grade<From||Grade>To);
	return Grade;
}

char GetGradeLetter(int Grade)
{
	if (Grade >= 90)
		return 'A';
	else if (Grade >= 80)
		return 'A';
	else if (Grade >= 70)
		return 'A';
	else if (Grade >= 60)
		return 'A';
	else if (Grade >= 50)
		return 'A';
	else
		return 'F';
}

int main()
{
	cout << endl << "result : " << GetGradeLetter(ReadNumberInRange(0, 100));
	return 0;
}
